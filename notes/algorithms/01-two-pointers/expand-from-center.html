def longest_palindrome(s):
if not s:
return ""
start, end = 0, 0
for i in range(len(s)):
# Odd-length palindrome
l1, r1 = expand(s, i, i)
if r1 - l1 > end - start:
start, end = l1, r1
# Even-length palindrome
l2, r2 = expand(s, i, i+1)
if r2 - l2 > end - start:
start, end = l2, r2
return s[start:end+1]
def expand(s, left, right):
while left >= 0 and right < len(s):
if s[left] == s[right]:
left -= 1
right += 1
else:
break
return left + 1, right - 1
