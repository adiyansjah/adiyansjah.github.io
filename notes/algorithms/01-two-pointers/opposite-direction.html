Algorithm
def two_sum(arr, target):
L, R = 0, len(arr) - 1
while L < R:
sum = arr[L] + arr[R]
if sum == target:
return [L, R] # Found!
elif sum < target:
L += 1 # Need larger
else:
R -= 1 # Need smaller
return None # Not found
